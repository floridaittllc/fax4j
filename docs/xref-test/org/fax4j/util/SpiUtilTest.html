49          faxJob.setTargetAddress("TEST_TARGET_ADDRESS");
50          faxJob.setTargetName("TEST_TARGET_NAME");
51          faxJob.setSenderName("TEST_SENDER_NAME");
52          faxJob.setSenderFaxNumber("TEST_SENDER_FAX_NUMBER");
53          faxJob.setSenderEmail("TEST_SENDER_EMAIL");
54          TemplateParameterEncoder encoder = null;
55          String output = SpiUtil.formatTemplate(template, faxJob, encoder, false, true);
56          Assert.assertNotNull(output);
57          Assert.assertNotSame(String.valueOf(-1), String.valueOf(output.indexOf("1:TEST_FILE_NAME\n")));
58          Assert.assertNotSame(String.valueOf(-1), String.valueOf(output.indexOf("2:TEST_TARGET_ADDRESS\n")));
59          Assert.assertNotSame(String.valueOf(-1), String.valueOf(output.indexOf("3:TEST_TARGET_NAME\n")));
