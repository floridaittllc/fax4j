197     /**
198      * Test
199      * 
200      * @throws Exception
201      *             Any exception
202      */
203     @Test
204     public void createHTTPRequestSubmitTest() throws Exception {
205         FaxJob faxJob = new FaxJobImpl();
206         faxJob.setFilePath("myfile.txt");
207         faxJob.setTargetAddress("TEST_TARGET_ADDRESS");
208         faxJob.setTargetName("TEST_TARGET_NAME");
209         faxJob.setSenderName("TEST_SENDER_NAME");
210         faxJob.setSenderFaxNumber("TEST_SENDER_FAX_NUMBER");
211         faxJob.setSenderEmail("TEST_SENDER_EMAIL");
212         HTTPRequest output = this.converter.createHTTPRequest(this.faxClientSpi, FaxActionType.SUBMIT_FAX_JOB, faxJob);
213 
214         Assert.assertNotNull(output);
215         Assert.assertEquals("param1=value1&param2=value2", output.getParametersText());
216         Assert.assertEquals("resource1", output.getResource());
217         ContentPart<?>[] parts = output.getContentAsParts();
218         Assert.assertNotNull(parts);
219         int foundPartsAmount = 0;
220         for (int index = 0; index < parts.length; index++) {
221             ContentPart<?> part = parts[index];
222             if (part.getName().equals(this.converter.submitFileContentParameter)) {
223                 Assert.assertEquals("myfile.txt", ((File) part.getContent()).getName());
224                 foundPartsAmount++;
225             } else if (part.getName().equals(this.converter.submitFileNameParameter)) {
226                 Assert.assertEquals("myfile.txt", part.getContent());
227                 foundPartsAmount++;
228             } else if (part.getName().equals(this.converter.submitTargetAddressParameter)) {
229                 Assert.assertEquals("TEST_TARGET_ADDRESS", part.getContent());
230                 foundPartsAmount++;
231             } else if (part.getName().equals(this.converter.submitTargetNameParameter)) {
232                 Assert.assertEquals("TEST_TARGET_NAME", part.getContent());
233                 foundPartsAmount++;
234             } else if (part.getName().equals(this.converter.submitSenderNameParameter)) {
235                 Assert.assertEquals("TEST_SENDER_NAME", part.getContent());
236                 foundPartsAmount++;
237             } else if (part.getName().equals(this.converter.submitSenderFaxNumberParameter)) {
238                 Assert.assertEquals("TEST_SENDER_FAX_NUMBER", part.getContent());
239                 foundPartsAmount++;
240             } else if (part.getName().equals(this.converter.submitSenderEMailParameter)) {
241                 Assert.assertEquals("TEST_SENDER_EMAIL", part.getContent());
242                 foundPartsAmount++;
243             } else if (part.getName().equals("KEY1")) {
244                 Assert.assertEquals("VALUE1", part.getContent());
245                 foundPartsAmount++;
246             } else if (part.getName().equals("KEY2")) {
247                 Assert.assertEquals("VALUE2", part.getContent());
248                 foundPartsAmount++;
249             }
250         }
251         Assert.assertEquals(9, foundPartsAmount);
252     }
