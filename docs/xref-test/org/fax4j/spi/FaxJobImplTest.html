3   import java.io.File;
4   import org.fax4j.FaxJob.FaxJobPriority;
5   import org.junit.Assert;

9   /**
10   * Test Class
11   * 
12   * @author Sagie Gur-Ari
13   */
14  public class FaxJobImplTest {
15      /** The fax job to test */
16      private FaxJobImpl faxJob;
17  
18      /**
19       * Sets up the SPI instance.
20       */
21      @Before
22      public void setUp() {
23          this.faxJob = new FaxJobImpl();
24          this.faxJob.setFile(new File("TEST_FILE"));
25          this.faxJob.setID("TEST_ID");
26          this.faxJob.setPriority(FaxJobPriority.LOW_PRIORITY);
27          this.faxJob.setProperty("prop1", "value1");
28          this.faxJob.setProperty("prop2", "value2");
29          this.faxJob.setSenderEmail("TEST_SENDER_EMAIL");
30          this.faxJob.setSenderName("TEST_SENDER_NAME");
31          this.faxJob.setSenderFaxNumber("TEST_SENDER_FAX_NUMBER");
32          this.faxJob.setTargetAddress("TEST_TARGET_ADDRESS");
33          this.faxJob.setTargetName("TEST_TARGET_NAME");
34      }
35  
36      /**
37       * Test
38       * 
39       * @throws Exception
40       *             Any exception
41       */
42      @Test
43      public void toStringTest() throws Exception {
44          String output = this.faxJob.toString();
45          Assert.assertNotNull(output);
46          Assert.assertNotSame(Integer.valueOf(-1), Integer.valueOf(output.indexOf("ID: TEST_ID")));
47          Assert.assertNotSame(Integer.valueOf(-1), Integer.valueOf(output.indexOf("Priority: LOW_PRIORITY")));
48          Assert.assertNotSame(Integer.valueOf(-1), Integer.valueOf(output.indexOf("Sender Email: TEST_SENDER_EMAIL")));
49          Assert.assertNotSame(Integer.valueOf(-1),
50                  Integer.valueOf(output.indexOf("Sender Fax Number: TEST_SENDER_FAX_NUMBER")));
51          Assert.assertNotSame(Integer.valueOf(-1),
52                  Integer.valueOf(output.indexOf("Target Address: TEST_TARGET_ADDRESS")));
53          Assert.assertNotSame(Integer.valueOf(-1), Integer.valueOf(output.indexOf("Target Name: TEST_TARGET_NAME")));
54          Assert.assertNotSame(Integer.valueOf(-1), Integer.valueOf(output.indexOf("Sender Name: TEST_SENDER_NAME")));
55          Assert.assertNotSame(Integer.valueOf(-1), Integer.valueOf(output.indexOf("File: TEST_FILE")));
56      }
57  }
