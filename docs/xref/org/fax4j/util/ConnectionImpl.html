1   package org.fax4j.util;
2   
3   import org.fax4j.FaxException;
4   
5   /**
6    * This class provides the basic common connection implementation.
7    * 
8    * @param <T>
9    *            The resource type
10   * @author Sagie Gur-Ari
11   * @version 1.01
12   * @since 0.39a
13   */
14  public class ConnectionImpl<T> extends AbstractCloseable implements Connection<T> {
15      /** The resource held by this connection */
16      private final T RESOURCE;
17      /** The owner connection factory */
18      private final ConnectionFactory<T> CONNECTION_FACTORY;
19  
20      /**
21       * This is the class constructor.
22       * 
23       * @param resource
24       *            The resource held by this connection
25       * @param connectionFactory
26       *            The owner connection factory
27       */
28      public ConnectionImpl(T resource, ConnectionFactory<T> connectionFactory) {
29          super();
30  
31          if (resource == null) {
32              throw new FaxException("Connection resource not provided.");
33          }
34          if (connectionFactory == null) {
35              throw new FaxException("Owner connection factory not provided.");
36          }
37  
38          // get resource
39          this.RESOURCE = resource;
40  
41          // get connection factory
42          this.CONNECTION_FACTORY = connectionFactory;
43      }
44  
45      /**
46       * Closes the connection.
47       */
48      @Override
49      protected final void closeImpl() {
50          this.CONNECTION_FACTORY.releaseConnection(this);
51      }
52  
53      /**
54       * This function returns the resource held by the connection.
55       * 
56       * @return The resource held by the connection
57       */
58      public final T getResource() {
59          return this.RESOURCE;
60      }
61  }
