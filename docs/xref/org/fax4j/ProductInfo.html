1   package org.fax4j;

3   import java.util.Properties;
4   import org.fax4j.common.Logger;
5   import org.fax4j.util.LibraryConfigurationLoader;
6   
7   /**
8    * This class is used to print the product info of the fax4j library.
9    * 
10   * @author Sagie Gur-Ari
11   * @version 1.05
12   * @since 0.1
13   */
14  public final class ProductInfo {
15      /**
16       * This is the class constructor.
17       */
18      private ProductInfo() {
19          super();
20      }
21  
22      /**
23       * Prints the product info to the system out and exists the process.
24       * 
25       * @param args
26       *            The command line arguments (not used)
27       */
28      public static void main(String[] args) {
29          // print product info
30          ProductInfo.printProductInfo();
31  
32          // exit
33          System.exit(0);
34      }
35  
36      /**
37       * Prints the product info to the system out.
38       */
39      public static void printProductInfo() {
40          // get product info
41          String text = ProductInfo.getProductInfoPrintout();
42  
43          // print values
44          System.out.println(text);
45      }
46  
47      /**
48       * Returns the product info printout.
49       * 
50       * @return The product info printout
51       */
52      public static String getProductInfoPrintout() {
53          // read properties
54          Properties properties = LibraryConfigurationLoader.readInternalConfiguration();
55  
56          // get values
57          String name = properties.getProperty("org.fax4j.product.name");
58          String version = properties.getProperty("org.fax4j.product.version");
59  
60          // init buffer
61          StringBuilder buffer = new StringBuilder();
62          buffer.append(name);
63          buffer.append(" library.");
64          buffer.append(Logger.SYSTEM_EOL);
65          buffer.append("Version: ");
66          buffer.append(version);
67  
68          // get text
69          String text = buffer.toString();
70  
71          return text;
72      }
73  }
