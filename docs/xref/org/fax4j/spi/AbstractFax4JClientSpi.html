9   /**
10   * This class provides partial/common functionlity of the internal fax4j fax client service provider interface.<br>
11   * This class should be used only by <b>internal</b> fax4j SPIs, for SPIs written by vendors outside the fax4j, see the
12   * org.fax4j.spi.AbstractFaxClientSpi class.<br>
13   * This is the actual engine behind the fax4j framework.<br>
14   * It handles the fax client activities such as sending/suspending fax jobs and so on.<br>
15   * The SPI is an internal class that should be used internally only.<br>
16   * <br>
17   * The configuration of the fax4j framework is made up of 3 layers.<br>
18   * The configuration is based on simple properties.<br>
19   * Each layer overrides the lower layers by adding/changing the property values.<br>
20   * The first layer is the internal fax4j.properties file located in the fax4j jar.<br>
21   * This layer contains the preconfigured values for the fax4j framework and can be changed by updating these properties
22   * in the higher layers.<br>
23   * The second layer is the external fax4j.properties file that is located on the classpath.<br>
24   * This file is optional and provides the ability to override the internal configuration for the entire fax4j
25   * framework.<br>
26   * The top most layer is the optional java.util.Properties object provided by the external classes when creating a new
27   * fax client.<br>
28   * These properties enable to override the configuration of the lower 2 layers.<br>
29   * <br>
30   * For SPI specific configuration, see the relevant SPI class javadoc.
31   *
32   * @author Sagie Gur-Ari
33   * @version 1.03
34   * @since 0.17
35   */

62      /**
63       * This function polls the new statues for the provided fax jobs.
64       *
65       * @param faxJobs
66       *            The fax jobs to poll
67       * @return The fax job statues
68       */
69      public final FaxJobStatus[] pollForFaxJobStatues(FaxJob[] faxJobs) {
70          FaxJobStatus[] faxJobStatuses = null;
71          if (faxJobs != null && faxJobs.length > 0 && this.isFaxMonitorEventsSupported()) {
72              faxJobStatuses = this.pollForFaxJobStatuesImpl(faxJobs);
73          }
74  
75          return faxJobStatuses;
76      }

138     /**
139      * This function polls the new statues for the provided fax jobs.
140      *
141      * @param faxJobs
142      *            The fax jobs to poll
143      * @return The fax job statues
144      */
145     protected FaxJobStatus[] pollForFaxJobStatuesImpl(FaxJob[] faxJobs) {
146         // get amount
147         int amount = faxJobs.length;
148 
149         // init array
150         FaxJobStatusbStatus[] faxJobStatuses = new FaxJobStatus[amount];
151 
152         FaxJob faxJob = null;
153         FaxJobStatus faxJobStatus = null;
154         final Logger logger = this.getLogger();
155         for (int index = 0; index < amount; index++) {
156             // get next fax job
157             faxJob = faxJobs[index];
158 
159             // set default
160             faxJobStatus = FaxJobStatus.UNKNOWN;
161 
162             if (faxJob != null) {
163                 try {
164                     faxJobStatus = this.getFaxJobStatus(faxJob);
165                 } catch (RuntimeException exception) {
166                     logger.logDebug(null, exception);
167                 }
168             }
169 
170             // set default
171             if (faxJobStatus == null) {
172                 faxJobStatus = FaxJobStatus.UNKNOWN;
173             }
174 
175             // put in output array
176             faxJobStatuses[index] = faxJobStatus;
177         }
178 
179         return faxJobStatuses;
180     }
