14   * This class provides the default implementation of the fax job monitor.

20  public class FaxJobMonitorImpl extends AbstractFaxJobMonitor {
21      /** The poller task interval between invocations */
22      protected long pollingInterval;
23      /** The fixed polling interval settings */
24      protected boolean fixedPollingInterval;
25      /** The poller thread priority */
26      protected int pollingThreadPriority;
27      /** The poller task */
28      protected PollerTask pollerTask;
29      /** The monitored fax jobs */
30      private Map<FaxClientSpi, Map<FaxJob, FaxJobStatus>> data;
31      /** The lock object */
32      protected final Object LOCK;
33      /** The fax job polling interval in millies property key */
34      public static final String POLLING_INTERVAL_IN_MILLIES_PROPERTY_KEY = "org.fax4j.monitor.polling.interval";
35      /** The fax job polling interval fixed property key */
36      public static final String FIXED_POLLING_INTERVAL_PROPERTY_KEY = "org.fax4j.monitor.polling.interval.fixed";
37      /** The fax job polling thread priority property key */
38      public static final String POLLING_THREAD_PRIORITY_PROPERTY_KEY = "org.fax4j.monitor.polling.thread.priority";

40      /**
41       * This is the class constructor.
42       */
43      public FaxJobMonitorImpl() {
44          super();
45  
46          // init lock
47          this.LOCK = new Object();
48      }

50      /**
51       * This function initializes the fax job monitor.
52       */
53      @Override
54      protected void initializeImpl() {
55          // init data structures
56          this.data = new HashMap<FaxClientSpi, Map<FaxJob, FaxJobStatus>>(20);
57  
58          // get polling interval
59          String value = this.getConfigurationValue(FaxJobMonitorImpl.POLLING_INTERVAL_IN_MILLIES_PROPERTY_KEY);
60          this.pollingInterval = 5000;
61          if (value != null) {
62              this.pollingInterval = Long.parseLong(value);
63              if (this.pollingInterval <= 0) {
64                  throw new FaxException("Polling interval set to an invalid value: " + this.pollingInterval);
65              }
66          }
67  
68          // get fixed polling interval flag
69          value = this.getConfigurationValue(FaxJobMonitorImpl.FIXED_POLLING_INTERVAL_PROPERTY_KEY);
70          this.fixedPollingInterval = Boolean.parseBoolean(value);
71  
72          // get poller thread priority
73          value = this.getConfigurationValue(FaxJobMonitorImpl.POLLING_THREAD_PRIORITY_PROPERTY_KEY);
74          this.pollingThreadPriority = 2;
75          if (value != null) {
76              this.pollingThreadPriority = Integer.parseInt(value);
77              if (this.pollingThreadPriority < Thread.MIN_PRIORITY || this.pollingThreadPriority > Thread.MAX_PRIORITY) {
78                  throw new FaxException(
79                          "Polling thread priority set to an invalid value: " + this.pollingThreadPriority);
80              }
81          }
82  
83          // create poller task
84          this.pollerTask = new PollerTask(this);
85      }

87      /**
88       * This function starts monitoring the requested fax job.
89       *
90       * @param faxClientSpi
91       *            The fax client SPI
92       * @param faxJob
93       *            The fax job to monitor
94       */
95      @Override
96      public void monitorFaxJobImpl(FaxClientSpi faxClientSpi, FaxJob faxJob) {
97          // get fax job status
98          FaxJobStatus faxJobStatus = faxClientSpi.getFaxJobStatus(faxJob);
99          if (faxJobStatus == null) {
100             throw new FaxException("Unable to extract fax job status for fax job: " + faxJob.getID());
101         }
102 
103         synchronized (this.LOCK) {
104             Map<FaxJob, FaxJobStatus> faxJobMap = this.data.get(faxClientSpi);
105             if (faxJobMap == null) {
106                 faxJobMap = new HashMap<FaxJob, FaxJobStatus>(500);
107                 this.data.put(faxClientSpi, faxJobMap);
108             }
109 
110             // add new fax job
111             faxJobMap.put(faxJob, faxJobStatus);
112 
113             if (!this.pollerTask.isRunning()) {
114                 this.pollerTask.startTask();
115             }
116         }
117     }

119     /**
120      * This function will stop the monitoring of all fax jobs registered by the provided fax client SPI.<br>
121      * Other fax jobs registered by other SPIs will keep on being monitored.
122      *
123      * @param faxClientSpi
124      *            The fax client SPI
125      */
126     @Override
127     public void stopMonitoringAllFaxJobsImpl(FaxClientSpi faxClientSpi) {
128         synchronized (this.LOCK) {
129             // remove from monitored data
130             this.data.remove(faxClientSpi);
131         }
132 
133         // check if need to stop the monitor
134         this.checkAndStopMonitor();
135     }

137     /**
138      * Runs the polling cycle to fetch and create monitor events.
139      */
140     protected void runPollingCycle() {
141         synchronized (this.LOCK) {
142             Set<Entry<FaxClientSpi, Map<FaxJob, FaxJobStatus>>> entrySet = this.data.entrySet();
143             Iterator<Entry<FaxClientSpi, Map<FaxJob, FaxJobStatus>>> faxClientSpiIterator = entrySet.iterator();
144             Entry<FaxClientSpi, Map<FaxJob, FaxJobStatus>> faxClientSpiEntry = null;
145             FaxClientSpi faxClientSpi = null;
146             Map<FaxJob, FaxJobStatus> faxJobMap = null;
147             Iterator<Entry<FaxJob, FaxJobStatus>> faxJobIterator = null;
148             Entry<FaxJob, FaxJobStatus> faxJobEntry = null;
149             FaxJob[] faxJobs = null;
150             FaxJobStatus[] previousFaxJobStatuses = null;
151             FaxJobStatus[] currentFaxJobStatuses = null;
152             FaxJob faxJob = null;
153             FaxJobStatus previousFaxJobStatus = null;
154             FaxJobStatus currentFaxJobStatus = null;
155             int counter = 0;
156             while (faxClientSpiIterator.hasNext()) {
157                 // get next entry
158                 faxClientSpiEntry = faxClientSpiIterator.next();
159 
160                 // get values
161                 faxClientSpi = faxClientSpiEntry.getKey();
162                 faxJobMap = faxClientSpiEntry.getValue();
163 
164                 // get iterator
165                 faxJobIterator = faxJobMap.entrySet().iterator();
166 
167                 int amount = faxJobMap.size();
168                 if (amount > 0) {
169                     // init arrays
170                     faxJobs = new FaxJob[amount];
171                     previousFaxJobStatuses = new FaxJobStatus[amount];
172 
173                     counter = 0;
174                     while (faxJobIterator.hasNext()) {
175                         // get next entry
176                         faxJobEntry = faxJobIterator.next();
177 
178                         // get values
179                         faxJob = faxJobEntry.getKey();
180                         previousFaxJobStatus = faxJobEntry.getValue();
181 
182                         // put in array
183                         faxJobs[counter] = faxJob;
184                         previousFaxJobStatuses[counter] = previousFaxJobStatus;
185                         counter++;
186                     }
187 
188                     // poll for changes
189                     currentFaxJobStatuses = faxClientSpi.pollForFaxJobStatues(faxJobs);
190 
191                     if (currentFaxJobStatuses != null && currentFaxJobStatuses.length == amount) {
192                         for (int index = 0; index < amount; index++) {
193                             // get current fax job status
194                             currentFaxJobStatus = currentFaxJobStatuses[index];
195 
196                             if (currentFaxJobStatus != null) {
197                                 // get values
198                                 faxJob = faxJobs[index];
199                                 previousFaxJobStatus = previousFaxJobStatuses[index];
200 
201                                 // if status was changed
202                                 if (!previousFaxJobStatus.equals(currentFaxJobStatus)) {
203                                     // update data structure
204                                     faxJobMap.put(faxJob, currentFaxJobStatus);
205 
206                                     // fire event
207                                     faxClientSpi.fireFaxMonitorEvent(FaxMonitorEventID.FAX_JOB_STATUS_CHANGE, faxJob,
208                                             currentFaxJobStatus);
209                                 }
210 
211                                 // if status is unknown or error, stop monitoring fax job
212                                 switch (currentFaxJobStatus) {
213                                 case UNKNOWN:
214                                 case ERROR:
215                                     faxJobMap.remove(faxJob);
216                                     break;
217                                 case IN_PROGRESS:
218                                 case PENDING:
219                                 default:
220                                     // do nothing
221                                     break;
222                                 }
223                             }
224                         }
225                     }
226                 }
227 
228                 if (faxJobMap.isEmpty()) {
229                     faxClientSpiIterator.remove();
230                 }
231             }
232         }
233 
234         // check if need to stop the monitor
235         this.checkAndStopMonitor();
236     }

238     /**
239      * Checks and if needed stops the monitor in case there is no data to monitor.
240      */
241     private void checkAndStopMonitor() {
242         synchronized (this.LOCK) {
243             // if no data, cancel poller task
244             if (this.data.isEmpty()) {
245                 this.pollerTask.setRunning(false);
246             }
247         }
248     }

250     /**
251      * This class used the start the polling cycle.
252      *
253      * @author Sagie Gur-Ari
254      * @version 1.02
255      * @since 0.40.2
256      */
257     public static class PollerTask implements Runnable {
258         /** The running flag */
259         private boolean running;
260         /** The fax job monitor */
261         private final FaxJobMonitorImpl FAX_JOB_MONITOR;
262 
263         /**
264          * This is the class constructor.
265          *
266          * @param faxJobMonitor
267          *            The fax job monitor
268          */
269         public PollerTask(FaxJobMonitorImpl faxJobMonitor) {
270             super();
271 
272             this.FAX_JOB_MONITOR = faxJobMonitor;
273             this.setRunning(false);
274         }
275 
276         /**
277          * This function sets the running flag.
278          *
279          * @param running
280          *            The running flag
281          */
282         public final void setRunning(boolean running) {
283             synchronized (this) {
284                 this.running = running;
285                 this.notifyAll();
286             }
287         }
288 
289         /**
290          * This function returns the running flag.
291          *
292          * @return The running flag
293          */
294         public final boolean isRunning() {
295             return this.running;
296         }
297 
298         /**
299          * Starts the task.
300          */
301         public final void startTask() {
302             if (!this.isRunning()) {
303                 synchronized (this) {
304                     // set flag
305                     this.setRunning(true);
306 
307                     // start thread
308                     Thread thread = new Thread(this, "Fax Monitor");
309                     thread.setPriority(this.FAX_JOB_MONITOR.pollingThreadPriority);
310                     thread.setDaemon(true);
311                     thread.start();
312                 }
313             }
314         }
315 
316         /**
317          * Invoked by the timer to start the polling cycle.
318          */
319         public final void run() {
320             // run loop
321             long startTime = -1;
322             long waitTime = 0;
323             long cycleTime = this.FAX_JOB_MONITOR.pollingInterval;
324             while (this.isRunning()) {
325                 // get start time
326                 startTime = System.currentTimeMillis();
327 
328                 // run internal runnable
329                 this.FAX_JOB_MONITOR.runPollingCycle();
330 
331                 // get wait time
332                 waitTime = cycleTime;
333                 if (!this.FAX_JOB_MONITOR.fixedPollingInterval) {
334                     waitTime = waitTime - (System.currentTimeMillis() - startTime);
335                 }
336 
337                 if (waitTime > 0) {
338                     synchronized (this) {
339                         try {
340                             // wait for next cycle
341                             this.wait(waitTime);
342                         } catch (Throwable throwable) { // NOPMD
343                             // ignore
344                         }
345                     }
346                 }
347             }
348         }
349     }
