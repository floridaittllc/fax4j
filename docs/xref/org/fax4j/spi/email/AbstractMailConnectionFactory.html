1   package org.fax4j.spi.email;

3   import java.util.Iterator;
4   import java.util.Map;
5   import java.util.Map.Entry;
6   import java.util.Properties;
7   import org.fax4j.spi.email.AbstractMailFaxClientSpi.FaxClientSpiConfigurationConstants;
8   import org.fax4j.util.AbstractFax4JConnectionFactory;
9   
10  /**
11   * Provides partial implementation of the mail connection factory.
12   *
13   * @author Sagie Gur-Ari
14   * @version 1.03
15   * @since 0.39a
16   */
17  public abstract class AbstractMailConnectionFactory extends AbstractFax4JConnectionFactory<MailResourcesHolder>
18          implements MailConnectionFactory {
19      /** The mail connection properties */
20      protected Properties mailConnectionProperties;
21      /** The mail connection user name */
22      protected String userName;
23      /** The mail connection password */
24      protected String password;
25      /** The transport protocol */
26      protected String transportProtocol;
27      /** The transport host */
28      protected String transportHost;
29      /** The transport port */
30      protected int transportPort;
31  
32      /**
33       * This is the class constructor.
34       */
35      public AbstractMailConnectionFactory() {
36          super();
37      }
38  
39      /**
40       * This function initializes the connection factory.
41       */
42      @Override
43      protected void initializeImpl() {
44          // create properties from configuration (all javax mail properties will be defined in the fax4j properties)
45          this.mailConnectionProperties = new Properties();
46          Map<String, String> configuration = this.factoryConfigurationHolder.getConfiguration();
47          Iterator<Entry<String, String>> iterator = configuration.entrySet().iterator();
48          Entry<String, String> entry = null;
49          String key = null;
50          String value = null;
51          while (iterator.hasNext()) {
52              // get next entry
53              entry = iterator.next();
54  
55              // get next key
56              key = entry.getKey();
57              if (key != null && key.indexOf("org.fax4j.") == -1) // filter out fax4j properties
58              {
59                  // get next value
60                  value = entry.getValue();
61  
62                  // put in properties
63                  this.mailConnectionProperties.setProperty(key, value);
64              }
65          }
66  
67          // get user/password values
68          this.userName = this.factoryConfigurationHolder
69                  .getConfigurationValue(FaxClientSpiConfigurationConstants.USER_NAME_PROPERTY_KEY);
70          this.password = this.factoryConfigurationHolder
71                  .getConfigurationValue(FaxClientSpiConfigurationConstants.PASSWORD_PROPERTY_KEY);
72  
73          // get transport protocol
74          this.transportProtocol = this.factoryConfigurationHolder.getConfigurationValue("mail.transport.protocol");
75          if (this.transportProtocol == null) {
76              this.transportProtocol = "smtp";
77          }
78  
79          // get transport host
80          this.transportHost = this.factoryConfigurationHolder.getConfigurationValue("mail.smtp.host");
81  
82          // get transport port
83          String transportPortStr = this.factoryConfigurationHolder.getConfigurationValue("mail.smtp.port");
84          this.transportPort = -1;
85          if (transportPortStr != null) {
86              this.transportPort = Integer.parseInt(transportPortStr);
87          }
88      }
89  }
