13  /**
14   * This class implements the fax client service provider interface.<br>
15   * This parial implementation will invoke the requests by sending emails to a mail server that supports conversion
16   * between email messages and fax messages.<br>
17   * The mail SPI supports persistent connection to enable to reuse the same connection for all fax operation invocations
18   * or to create a new connection for each fax operation invocation.<br>
19   * By default the SPI will create a new connection for each operation invocation however the
20   * <b>org.fax4j.spi.mail.persistent.connection</b> set to true will enable to reuse the connection.<br>
21   * To set the user/password values of the mail connection the following 2 properties must be defined:
22   * <b>org.fax4j.spi.mail.user.name</b> and <b>org.fax4j.spi.mail.password</b><br>
23   * All properties defined in the fax4j configuration will be passed to the mail connection therefore it is possible to
24   * define mail specific properties (see java mail for more info) in the fax4j properties.<br>
25   * Implementing SPI class will have to implement the createXXXFaxJobMessage methods.<br>
26   * These methods will return the message to be sent for that fax job operation. In case the method returns null, this
27   * class will throw an UnsupportedOperationException exception. <br>
28   * The configuration of the fax4j framework is made up of 3 layers.<br>
29   * The configuration is based on simple properties.<br>
30   * Each layer overrides the lower layers by adding/changing the property values.<br>
31   * The first layer is the internal fax4j.properties file located in the fax4j jar.<br>
32   * This layer contains the preconfigured values for the fax4j framework and can be changed by updating these properties
33   * in the higher layers.<br>
34   * The second layer is the external fax4j.properties file that is located on the classpath.<br>
35   * This file is optional and provides the ability to override the internal configuration for the entire fax4j
36   * framework.<br>
37   * The top most layer is the optional java.util.Properties object provided by the external classes when creating a new
38   * fax client.<br>
39   * These properties enable to override the configuration of the lower 2 layers.<br>
40   * <br>
41   * <b>SPI Status (Draft, Beta, Stable): </b>Stable<br>
42   * <br>
43   * Below table describes the configuration values relevant for this class.<br>

269     /**
270      * Returns the mail connection to be used to send the fax via mail.
271      *
272      * @return The mail connection
273      */
274     protected Connection<MailResourcesHolder> getMailConnection() {
275         Connection<MailResourcesHolder> mailConnection = null;
276         if (this.usePersistentConnection) {
277             synchronized (this) {
278                 if (this.connection == null) {
279                     // create new connection
280                     this.connection = this.createMailConnection();
281                 }
282             }
283 
284             // get connection
285             mailConnection = this.connection;
286         } else {
287             // create new connection
288             mailConnection = this.createMailConnection();
289         }
290 
291         return mailConnection;
292     }

294     /**
295      * This function will send the mail message.
296      *
297      * @param faxJob
298      *            The fax job object containing the needed information
299      * @param mailConnection
300      *            The mail connection (will be released if not persistent)
301      * @param message
302      *            The message to send
303      */
304     protected void sendMail(FaxJob faxJob, Connection<MailResourcesHolder> mailConnection, Message message) {
305         if (message == null) {
306             this.throwUnsupportedException();
307         } else {
308             // get holder
309             MailResourcesHolder mailResourcesHolder = mailConnection.getResource();
310 
311             // get transport
312             Transport transport = mailResourcesHolder.getTransport();
313 
314             try {
315                 // send message
316                 message.saveChanges();
317                 if (transport == null) {
318                     Transport.send(message, message.getAllRecipients());
319                 } else {
320                     transport.sendMessage(message, message.getAllRecipients());
321                 }
322             } catch (Throwable throwable) {
323                 throw new FaxException("Unable to send message.", throwable);
324             } finally {
325                 if (!this.usePersistentConnection) {
326                     try {
327                         // close connection
328                         this.closeMailConnection(mailConnection);
329                     } catch (Exception exception) {
330                         // log error
331                         Logger logger = this.getLogger();
332                         logger.logInfo(new Object[] { "Error while releasing mail connection." }, exception);
333                     }
334                 }
335             }
336         }
337     }
