5   import org.fax4j.spi.FaxClientSpi;

8   /**
9    * This process output validator simply checks the exit code is 0.
10   * 
11   * @author Sagie Gur-Ari
12   * @version 1.0
13   * @since 0.26
14   */
15  public class ExitCodeProcessOutputValidator implements ProcessOutputValidator {
16      /**
17       * This is the class constructor.
18       */
19      public ExitCodeProcessOutputValidator() {
20          super();
21      }
22  
23      /**
24       * This function validates the process output for errors.<br>
25       * If not valid, an exception should be thrown.
26       * 
27       * @param faxClientSpi
28       *            The fax client SPI
29       * @param processOutput
30       *            The process output to validate
31       * @param faxActionType
32       *            The fax action type
33       */
34      public void validateProcessOutput(FaxClientSpi faxClientSpi, ProcessOutput processOutput,
35              FaxActionType faxActionType) {
36          // validate exit code
37          int exitCode = processOutput.getExitCode();
38          if (exitCode != 0) {
39              throw new FaxException(
40                      "Error while invoking fax action process, exit code: " + exitCode + " process output:\n"
41                              + processOutput.getOutputText() + "\nprocess error:\n" + processOutput.getErrorText());
42          }
43      }
44  }
