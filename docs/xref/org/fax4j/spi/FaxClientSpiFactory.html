224         // get layered configuration

241             throw new FaxException("Class name for type: " + updatedType + " not configured in fax4j.properties");
242         }
243 
244         // get logger
245         Logger spiLogger = FaxClientSpiFactory.getLogger();
246 
247         // get fax job monitor
248         FaxJobMonitor spiFaxJobMonitor = FaxClientSpiFactory.getFaxJobMonitor();

289     /**
290      * Returns the configuration to be used by the new SPI instance.<br>
291      * The configuration of the fax client SPI is made up of 3 layers. The lowest layer is the internal fax4j.properties
292      * file located in the fax4j jar. This configuration file contains the initial default configuration. The next layer
293      * is the external fax4j.properties file that is located on the classpath. This file is optional and provides the
294      * ability to override the internal configuration. The top most layer is the optional properties object provided by
295      * the external classes. These properties enable to override the configuration of the lower 2 layers.
296      *
297      * @param configuration
298      *            The fax client configuration (may be null)
299      * @return The fax client layered configuration
300      */
301     protected static Map<String, String> getConfiguration(Properties configuration) {
302         // get system configuration
303         Map<String, String> systemConfig = LibraryConfigurationLoader.getSystemConfiguration();
304 
305         // create new map
306         Map<String, String> layeredConfiguration = new HashMap<String, String>(systemConfig);
307         if (configuration != null) {
308             // convert to map
309             SpiUtil.copyPropertiesToMap(configuration, layeredConfiguration);
310         }
311 
312         return layeredConfiguration;

425     private static FaxClientSpiInterceptor[] createFaxClientSpiInterceptors(FaxClientSpi faxClientSpi) {
426         // get type list
427         String typeListStr = faxClientSpi.getConfigurationValue(FaxClientSpiFactory.SPI_INTERCEPTOR_LIST_PROPERTY_KEY);
428         FaxClientSpiInterceptor[] interceptors = null;
429         if (typeListStr != null) {
430             // split list
431             String[] types = typeListStr.split(FaxClientSpiFactory.SPI_INTERCEPTOR_LIST_SEPARATOR);
432 
433             // get amount
434             int amount = types.length;
435 
436             String type = null;
437             String propertyKey = null;
438             String className = null;
439             FaxClientSpiInterceptor interceptor = null;
440             List<FaxClientSpiInterceptor> interceptorList = new LinkedList<FaxClientSpiInterceptor>();
441             for (int index = 0; index < amount; index++) {
442                 // get next type
443                 type = types[index];
444 
445                 if (type != null && type.length() > 0) {
446                     // get property key
447                     propertyKey = FaxClientSpiFactory.SPI_INTERCEPTOR_TYPE_PROPERTY_KEY_PREFIX + type;
448 
449                     // get class name
450                     className = faxClientSpi.getConfigurationValue(propertyKey);
451 
452                     if (className == null) {
453                         throw new FaxException(
454                                 "Class name not defined by property: " + propertyKey + " for SPI interceptor.");
455                     }
456 
457                     // create new instance
458                     interceptor = (FaxClientSpiInterceptor) ReflectionHelper.createInstance(className);
459 
460                     // initialize
461                     interceptor.initialize(faxClientSpi);
462 
463                     // add to list
464                     interceptorList.add(interceptor);
465                 }
466 
467                 // convert to array
468                 interceptors = interceptorList.toArray(new FaxClientSpiInterceptor[interceptorList.size()]);
