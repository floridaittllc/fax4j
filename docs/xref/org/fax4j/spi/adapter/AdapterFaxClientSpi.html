13  /**
14   * This class implements the fax client service provider interface.<br>
15   * This implementation will proxy the requests to another internal SPI.<br>
16   * The internal SPI created will be based on configuration in the fax4j properties.<br>
17   * The configuration contains conditions such as OS, availability of java classes, availability of native libraries,
18   * availability of executables or validation that a certain fax4j property is defined and contains a value.<br>
19   * Once all conditions are met the internal SPI is created, otherwise the next SPI configuration is used.<br>
20   * If no SPI conditions are met, an exception will be thrown.<br>
21   * <br>
22   * The configuration of the fax4j framework is made up of 3 layers.<br>
23   * The configuration is based on simple properties.<br>
24   * Each layer overrides the lower layers by adding/changing the property values.<br>
25   * The first layer is the internal fax4j.properties file located in the fax4j jar.<br>
26   * This layer contains the preconfigured values for the fax4j framework and can be changed by updating these properties
27   * in the higher layers.<br>
28   * The second layer is the external fax4j.properties file that is located on the classpath.<br>
29   * This file is optional and provides the ability to override the internal configuration for the entire fax4j
30   * framework.<br>
31   * The top most layer is the optional java.util.Properties object provided by the external classes when creating a new
32   * fax client.<br>
33   * These properties enable to override the configuration of the lower 2 layers.<br>
34   * <br>

219      * This function initializes the fax client SPI.

221     @Override
222     protected void initializeChildFaxClientSpi() {
223         // get logger
224         Logger logger = this.getLogger();
225 
226         // get types
227         String types = this.getConfigurationValue(FaxClientSpiConfigurationConstants.SPI_TYPES_PROPERTY_KEY);
228         logger.logDebug(new Object[] { FaxClientSpiConfigurationConstants.SPI_TYPES_PROPERTY_KEY, " value is ", types },
229                 null);
230 
231         if (types != null) {
232             // split type values
233             String[] typeValues = types
234                     .split(FaxClientSpiConfigurationConstants.PROPERTY_VALUE_LIST_SEPARATOR.toString());
235 
236             // get amount
237             int amount = typeValues.length;
238 
239             String typeValue = null;
240             String conditionString = null;
241             String[] conditionBlocks = null;
242             String conditionBlock = null;
243             String[] conditionBlockValues = null;
244             String conditionKey = null;
245             String conditionValue = null;
246             int subAmount = -1;
247             boolean conditionValid = true;
248             StringBuilder conditionStringBuffer = new StringBuilder();
249             for (int index = 0; index < amount; index++) {
250                 // get next element
251                 typeValue = typeValues[index];
252                 logger.logDebug(new Object[] { "Checking conditions for type: ", typeValue }, null);
253 
254                 // get condition string
255                 conditionString = this.getConfigurationValue(
256                         FaxClientSpiConfigurationConstants.SPI_CONDITION_PROPERTY_KEY_PREFIX + typeValue);
257                 logger.logDebug(
258                         new Object[] { "Configured condition for type: ", typeValue, " is set to: ", conditionString },
259                         null);
260 
261                 // append mandatory conditions
262                 conditionStringBuffer.delete(0, conditionStringBuffer.length());
263                 conditionStringBuffer.append(FaxClientSpiConfigurationConstants.PROPERTY_CONDITION);
264                 conditionStringBuffer.append(FaxClientSpiConfigurationConstants.PROPERTY_BLOCK_VALUE_SEPARATOR);
265                 conditionStringBuffer.append(FaxClientSpiFactory.SPI_CLASS_NAME_TYPE_MAP_PROPERTY_KEY_PREFIX);
266                 conditionStringBuffer.append(typeValue);
267                 conditionStringBuffer.append(FaxClientSpiConfigurationConstants.PROPERTY_VALUE_LIST_SEPARATOR);
268                 conditionStringBuffer.append(FaxClientSpiConfigurationConstants.JAVA_CLASS_CONDITION);
269                 conditionStringBuffer.append(FaxClientSpiConfigurationConstants.PROPERTY_BLOCK_VALUE_SEPARATOR);
270                 conditionStringBuffer.append(this.getConfigurationValue(
271                         FaxClientSpiFactory.SPI_CLASS_NAME_TYPE_MAP_PROPERTY_KEY_PREFIX + typeValue));
272                 conditionStringBuffer.append(FaxClientSpiConfigurationConstants.PROPERTY_VALUE_LIST_SEPARATOR);
273                 conditionStringBuffer.append(FaxClientSpiConfigurationConstants.STABLE_CONDITION);
274                 conditionStringBuffer.append(FaxClientSpiConfigurationConstants.PROPERTY_BLOCK_VALUE_SEPARATOR);
275                 conditionStringBuffer.append(typeValue);
276                 if (conditionString != null) {
277                     conditionStringBuffer.append(FaxClientSpiConfigurationConstants.PROPERTY_VALUE_LIST_SEPARATOR);
278                     conditionStringBuffer.append(conditionString);
279                 }
280                 conditionString = conditionStringBuffer.toString();
281                 logger.logDebug(
282                         new Object[] { "Updated condition for type: ", typeValue, " is set to: ", conditionString },
283                         null);
284 
285                 // split condition to blocks
286                 conditionBlocks = conditionString
287                         .split(FaxClientSpiConfigurationConstants.PROPERTY_VALUE_LIST_SEPARATOR.toString());
288 
289                 // get amount
290                 subAmount = conditionBlocks.length;
291 
292                 conditionValid = true;
293                 for (int blockIndex = 0; blockIndex < subAmount; blockIndex++) {
294                     // get next element
295                     conditionBlock = conditionBlocks[blockIndex];
296 
297                     // split condition block
298                     conditionBlockValues = conditionBlock
299                             .split(FaxClientSpiConfigurationConstants.PROPERTY_BLOCK_VALUE_SEPARATOR.toString());
300 
301                     // check condition block contains only 2 parts (key and value)
302                     if (conditionBlockValues.length != 2) {
303                         throw new FaxException("Condition block must contain 2 parts separated by "
304                                 + FaxClientSpiConfigurationConstants.PROPERTY_BLOCK_VALUE_SEPARATOR);
305                     }
306 
307                     // get condition key and value
308                     conditionKey = conditionBlockValues[0];
309                     conditionValue = conditionBlockValues[1];
310 
311                     // validate condition
312                     logger.logDebug(new Object[] { "Validating condition for type: ", typeValue, " key: ", conditionKey,
313                             " value: ", conditionValue }, null);
314                     if (!this.validateCondition(conditionKey, conditionValue)) {
315                         conditionValid = false;
316                         break;
317                     }
318                 }
319 
320                 if (conditionValid) {
321                     logger.logDebug(new Object[] { "Conditions validated for type: ", typeValue }, null);
322 
323                     // create SPI
324                     this.createFaxClientSpi(typeValue);
325 
326                     break;
327                 }
328             }
329         }
330 
331         if (this.faxClientSpi == null) {
332             throw new FaxException("No internal SPI available.");
333         }
334     }

336     /**
337      * Validates the SPI condition.
338      *
339      * @param key
340      *            The condition key <br>
341      *            property - checks a fax4j property is defined and contains a value<br>
342      *            OS - checks OS name contains the value (case insensitive)<br>
343      *            java-class - checks that java class can be loaded<br>
344      *            native-lib - checks that native lib can be loaded<br>
345      *            executable - checks executable is on system path<br>
346      *            stable - checks for a fax4j property org.fax4j.spi.xxx.stable is defined and equals true, where xxx is
347      *            the SPI key (for example: org.fax4j.spi.adapter.stable=true)
348      * @param value
349      *            The condition value
350      * @return True if the condition is met
351      */
352     protected boolean validateCondition(String key, String value) {
353         boolean valid = false;
354         FaxClientSpiConfigurationConstants condition = FaxClientSpiConfigurationConstants.getEnum(key);
355         String propertyValue = null;
356         switch (condition) {
357         case PROPERTY_CONDITION:
358             // get value
359             propertyValue = this.getConfigurationValue(value);
360 
361             if (propertyValue != null) {
362                 valid = true;
363             }
364             break;
365         case OS_CONDITION:
366             // get OS name
367             String osName = System.getProperty("os.name");
368 
369             // change values to lower case
370             osName = osName.toLowerCase();
371             String updatedValue = value.toLowerCase();
372 
373             if (osName.contains(updatedValue)) {
374                 valid = true;
375             }
376             break;
377         case JAVA_CLASS_CONDITION:
378             try {
379                 // load class
380                 ReflectionHelper.getType(value);
381 
382                 valid = true;
383             } catch (Throwable throwable) { // NOPMD
384                 // ignore
385             }
386             break;
387         case NATIVE_LIB_CONDITION:
388             try {
389                 // load library
390                 System.loadLibrary(value);
391 
392                 valid = true;
393             } catch (Throwable throwable) { // NOPMD
394                 // ignore
395             }
396             break;
397         case EXECUTABLE_CONDITION:
398             // get file
399             File file = new File(value);
400 
401             if (file.exists() && file.isFile()) {
402                 valid = true;
403             } else {
404                 // get system patOSh
405                 String systemPath = System.getProperty("java.library.path");
406                 String[] systemPathElements = systemPath.split(System.getProperty("path.separator"));
407 
408                 // get amount
409                 int amount = systemPathElements.length;
410 
411                 String directoryPath = null;
412                 for (int index = 0; index < amount; index++) {
413                     // get next element
414                     directoryPath = systemPathElements[index];
415 
416                     // get file
417                     file = new File(directoryPath, value);
418 
419                     if (file.exists() && file.isFile()) {
420                         valid = true;
421                         break;
422                     }
423                 }
424             }
425             break;
426         case STABLE_CONDITION:
427             // get property key
428             String propertyKey = FaxClientSpiConfigurationConstants.SPI_STABLE_PROPERTY_KEY_PREFIX + value
429                     + FaxClientSpiConfigurationConstants.SPI_STABLE_PROPERTY_KEY_SUFFIX;
430 
431             // get value
432             propertyValue = this.getConfigurationValue(propertyKey);
433 
434             valid = Boolean.parseBoolean(propertyValue);
435             break;
436         default:
437             throw new FaxException("Invalid condition key provided: " + key);
438         }
439 
440         return valid;
441     }

457     /**
458      * This function creates and returns the internal fax client SPI configuration.
459      *
460      * @return The internal fax client SPI configuration
461      */
462     protected Properties createFaxClientSpiConfiguration() {
463         // get configuration
464         Map<String, String> configuration = this.getConfiguration();
465 
466         // create new configuration
467         Properties internalSPIConfiguration = new Properties();
468         Properties overrideConfiguration = new Properties();
469 
470         // go over current configuration
471         Iterator<Entry<String, String>> iterator = configuration.entrySet().iterator();
472         Entry<String, String> entry = null;
473         String key = null;
474         String value = null;
475         String overrideKey = null;
476         String overridePrefix = FaxClientSpiConfigurationConstants.SPI_CONFIGURATION_OVERRIDE_PROPERTY_KEY_PREFIX
477                 .toString();
478         int prefixLength = overridePrefix.length();
479         while (iterator.hasNext()) {
480             // get next entry
481             entry = iterator.next();
482 
483             // get key/value
484             key = entry.getKey();
485             value = entry.getValue();
486 
487             // override internal configuration
488             if (key.startsWith(overridePrefix) && key.length() != prefixLength) {
489                 overrideKey = key.substring(prefixLength);
490 
491                 // put in override configuration
492                 overrideConfiguration.setProperty(overrideKey, value);
493             }
494 
495             // put in new configuration
496             internalSPIConfiguration.setProperty(key, value);
497         }
498 
499         // override configuration
500         internalSPIConfiguration.putAll(overrideConfiguration);
501 
502         return internalSPIConfiguration;
503     }
