1   package org.fax4j;
2   
3   import java.util.EventListener;
4   
5   /**
6    * This interface defines the fax client action event listener.<br>
7    * The listener picks up events based on actions invoked on the fax client and not based on the status changes of the
8    * actual fax job.<br>
9    * For example is a fax job is submitted via fax client, than the faxJobSubmitted event method will be invoked.<br>
10   * However the fax job itself might still be pending transmission by the actual fax modem/server.
11   *
12   * @author Sagie Gur-Ari
13   * @version 1.02
14   * @since 0.20b
15   */
16  public interface FaxClientActionEventListener extends EventListener {
17      /**
18       * This function is invoked after a new fax job is created.
19       *
20       * @param event
21       *            The fax event
22       */
23      void faxJobCreated(FaxClientActionEvent event);
24  
25      /**
26       * This function is invoked after a fax job is submitted.
27       *
28       * @param event
29       *            The fax event
30       */
31      void faxJobSubmitted(FaxClientActionEvent event);
32  
33      /**
34       * This function is invoked after a fax job is suspended.
35       *
36       * @param event
37       *            The fax event
38       */
39      void faxJobSuspended(FaxClientActionEvent event);
40  
41      /**
42       * This function is invoked after a fax job is resumed.
43       *
44       * @param event
45       *            The fax event
46       */
47      void faxJobResumed(FaxClientActionEvent event);
48  
49      /**
50       * This function is invoked after a fax job is cancelled.
51       *
52       * @param event
53       *            The fax event
54       */
55      void faxJobCancelled(FaxClientActionEvent event);
56  }
